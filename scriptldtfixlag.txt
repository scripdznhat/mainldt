local dragging, dragInput, dragStart, startPos
TitleBar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = MainFrame.Position
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then dragging = false end
        end)
    end
end)
TitleBar.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then dragInput = input end
end)
UserInputService.InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        local delta = input.Position - dragStart
        MainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)

-- LÕI XỬ LÝ FIX LAG (CHUYÊN TRỊ BLOX FRUITS)

local CurrentLevel = 0

-- Tối ưu hóa hệ thống ánh sáng và nước biển
local function OptimizeLightingAndTerrain()
    lighting.GlobalShadows = false
    lighting.FogEnd = 9e9
    lighting.Brightness = 1
    
    -- Xóa sương mù, mờ viền, chói sáng của Blox Fruits
    for _, v in pairs(lighting:GetChildren()) do
        if v:IsA("PostEffect") or v:IsA("Atmosphere") or v:IsA("Sky") or v:IsA("ColorCorrectionEffect") or v:IsA("BloomEffect") or v:IsA("BlurEffect") or v:IsA("SunRaysEffect") then
            v:Destroy()
        end
    end

    -- Giảm đồ họa nước biển
    if terrain then
        terrain.WaterWaveSize = 0
        terrain.WaterWaveSpeed = 0
        terrain.WaterReflectance = 0
        terrain.WaterTransparency = 1
    end
end

-- Hàm xử lý từng vật thể (Quét sâu vào MeshPart)
local function ProcessObject(obj, level)
    -- LEVEL 1: Tắt bóng và ánh sáng phát ra
    if level >= 1 then
        if obj:IsA("BasePart") then
            obj.CastShadow = false
        elseif obj:IsA("PointLight") or obj:IsA("SurfaceLight") or obj:IsA("SpotLight") then
            obj.Enabled = false
        end
    end

    -- LEVEL 2: Xóa sổ 100% hiệu ứng chiêu thức (Particles, Trails)
    if level >= 2 then
        if obj:IsA("ParticleEmitter") or obj:IsA("Trail") or obj:IsA("Beam") or obj:IsA("Fire") or obj:IsA("Smoke") or obj:IsA("Sparkles") then
            obj.Enabled = false -- Chỉ tắt đi, không Destroy để tránh lỗi script game
            obj.Transparency = NumberSequence.new(1) -- Ép tàng hình
        end
    end

    -- LEVEL 3: Tàn phá đồ họa (Xóa Texture, làm trong suốt)
    if level >= 3 then
        if obj:IsA("BasePart") then
            obj.Material = Enum.Material.SmoothPlastic
            obj.Color = Color3.fromRGB(150, 150, 150)
            obj.Transparency = 0.5 -- Nhìn xuyên thấu
        end
        
        -- ĐẶC BIỆT CHO BLOX FRUITS: Xóa kết cấu Mesh
        if obj:IsA("MeshPart") then
            obj.TextureID = "" -- Xóa vân bề mặt
        elseif obj:IsA("SpecialMesh") then
            obj.TextureId = "" -- Xóa ảnh của các khối custom
        elseif obj:IsA("Decal") or obj:IsA("Texture") then
            obj.Transparency = 1 -- Làm mờ hoàn toàn ảnh dán
        end
    end
end

-- Quét toàn bộ Game (Tránh lag máy khi quét bằng cách chia nhỏ bằng task.wait)
local function RunOptimizer(level)
    CurrentLevel = level
    OptimizeLightingAndTerrain()
    
    -- Quét cả Workspace (Đảo, NPC) và ReplicatedStorage (Chiêu thức đang cất giấu)
    local allObjects = workspace:GetDescendants()
    for i, obj in pairs(allObjects) do
        ProcessObject(obj, level)
        -- Chống crash khi quét quá nhiều vật thể cùng lúc
        if i % 1000 == 0 then task.wait() end 
    end
end

-- Tự động áp dụng lên các chiêu thức VỪA ĐƯỢC TUNG RA
workspace.DescendantAdded:Connect(function(obj)
    if CurrentLevel > 0 then
        ProcessObject(obj, CurrentLevel)
    end
end)
