local dragging, dragInput, dragStart, startPos
TitleBar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true; dragStart = input.Position; startPos = MainFrame.Position
        input.Changed:Connect(function() if input.UserInputState == Enum.UserInputState.End then dragging = false end end)
    end
end)

    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then dragInput = input end
end)

    if input == dragInput and dragging then
        local delta = input.Position - dragStart
        MainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)

-- =======================================================
-- LÕI XỬ LÝ FIX LAG (CÓ LỚP BẢO VỆ CHỐNG LỖI)
-- =======================================================
local CurrentLevel = 0
local isProcessing = false 

local function OptimizeLightingAndTerrain()
    pcall(function()
        lighting.GlobalShadows = false
        lighting.FogEnd = 9e9
        
        for _, v in pairs(lighting:GetChildren()) do
            if v:IsA("PostEffect") or v:IsA("Atmosphere") or v:IsA("Sky") or v:IsA("ColorCorrectionEffect") or v:IsA("BloomEffect") or v:IsA("SunRaysEffect") then
                v:Destroy()
            end
        end

        if terrain then
            terrain.WaterWaveSize = 0; terrain.WaterWaveSpeed = 0; terrain.WaterReflectance = 0; terrain.WaterTransparency = 1
            if CurrentLevel >= 2 then terrain.Decoration = false end
        end
    end)
end

local function ProcessObject(obj, level)
    -- Lệnh pcall sẽ giúp ngăn chặn kẹt nút nếu vật thể bị game khóa
    pcall(function()
        if level >= 1 then
            if obj:IsA("BasePart") then obj.CastShadow = false end
            if obj:IsA("Light") then obj.Enabled = false end
        end

        if level >= 2 then
            if obj:IsA("ParticleEmitter") or obj:IsA("Trail") or obj:IsA("Beam") or obj:IsA("Fire") or obj:IsA("Smoke") or obj:IsA("Sparkles") then
                obj.Enabled = false
            end
        end

        if level >= 3 then
            if obj:IsA("BasePart") then
                obj.Material = Enum.Material.SmoothPlastic
                obj.Color = Color3.fromRGB(150, 150, 150)
                obj.Transparency = 0.5
                obj.Reflectance = 0
            end
            if obj:IsA("MeshPart") then obj.TextureID = "" end
            if obj:IsA("SpecialMesh") then obj.TextureId = "" end
            if obj:IsA("Decal") or obj:IsA("Texture") then obj.Transparency = 1 end
            
            -- Đổi từ Destroy() sang giấu đi để an toàn không lỗi game
            if obj:IsA("Accessory") or obj:IsA("ShirtGraphic") or obj:IsA("Shirt") or obj:IsA("Pants") then
                obj:Destroy()
            end
        end
    end)
end

-- =======================================================
-- TỐI ƯU HÓA HÀM CHẠY (BẤT TỬ, KHÔNG THỂ BỊ KẸT)
-- =======================================================
local function RunOptimizer(level, btn, doneText)
    if isProcessing then return end 
    isProcessing = true
    CurrentLevel = level
    
    btn.Text = "Đang áp dụng..."
    
    task.spawn(function()
        -- Cho dù lỗi ở đâu, vẫn sẽ mở khóa nút
        local success, err = pcall(function()
            OptimizeLightingAndTerrain()
            
            local objects = workspace:GetDescendants()
            local count = 0
            
            for i = 1, #objects do
                ProcessObject(objects[i], level)
                count = count + 1
                if count >= 3000 then -- Cứ 3000 vật thể chớp mắt 1 lần để mượt
                    count = 0
                    RunService.Heartbeat:Wait()
                end
            end
        end)
        
        -- Dù thành công hay lỗi, luôn luôn thực hiện những dòng sau:
        btn.Text = doneText
        isProcessing = false -- Mở khóa nút bấm!
        
        if not success then
            warn("Đã fix lag nhưng có vài vật bị khóa: " .. tostring(err))
        end
    end)
end

-- Tự động bắt hiệu ứng chiêu mới quăng ra
workspace.DescendantAdded:Connect(function(obj)
    if CurrentLevel > 0 then
        task.defer(function()
            ProcessObject(obj, CurrentLevel)
        end)
    end
end)
