-- =======================================================
-- HỆ THỐNG LƯU TRỮ VÀ XỬ LÝ (CACHE SYSTEM)
-- =======================================================
local CurrentLevel = 0

-- Hàm LƯU trạng thái gốc trước khi thay đổi
local function SaveOriginalState(obj)
    if obj:GetAttribute("Saved") then return end -- Nếu đã lưu rồi thì bỏ qua
    obj:SetAttribute("Saved", true)

    if obj:IsA("BasePart") then
        obj:SetAttribute("OrigColor", obj.Color)
        obj:SetAttribute("OrigMaterial", obj.Material.Name)
        obj:SetAttribute("OrigTransparency", obj.Transparency)
        obj:SetAttribute("OrigShadow", obj.CastShadow)
    end
    if obj:IsA("MeshPart") then obj:SetAttribute("OrigTex", obj.TextureID) end
    if obj:IsA("SpecialMesh") then obj:SetAttribute("OrigTexId", obj.TextureId) end
    if obj:IsA("Decal") or obj:IsA("Texture") then obj:SetAttribute("OrigTrans", obj.Transparency) end
    if obj:IsA("ParticleEmitter") or obj:IsA("Trail") or obj:IsA("Beam") or obj:IsA("Light") or obj:IsA("PostEffect") then
        obj:SetAttribute("OrigEnabled", obj.Enabled)
    end
    if obj:IsA("Sky") then
        obj:SetAttribute("OrigSkyParent", obj.Parent.Name) -- Lưu lại để biết nó từng ở đâu
    end
end

-- Hàm KHÔI PHỤC trạng thái gốc
local function RestoreObject(obj)
    if not obj:GetAttribute("Saved") then return end

    if obj:IsA("BasePart") then
        if obj:GetAttribute("OrigColor") then obj.Color = obj:GetAttribute("OrigColor") end
        if obj:GetAttribute("OrigMaterial") then obj.Material = Enum.Material[obj:GetAttribute("OrigMaterial")] end
        if obj:GetAttribute("OrigTransparency") then obj.Transparency = obj:GetAttribute("OrigTransparency") end
        if obj:GetAttribute("OrigShadow") ~= nil then obj.CastShadow = obj:GetAttribute("OrigShadow") end
    end
    if obj:IsA("MeshPart") and obj:GetAttribute("OrigTex") then obj.TextureID = obj:GetAttribute("OrigTex") end
    if obj:IsA("SpecialMesh") and obj:GetAttribute("OrigTexId") then obj.TextureId = obj:GetAttribute("OrigTexId") end
    if (obj:IsA("Decal") or obj:IsA("Texture")) and obj:GetAttribute("OrigTrans") then obj.Transparency = obj:GetAttribute("OrigTrans") end
    if (obj:IsA("ParticleEmitter") or obj:IsA("Trail") or obj:IsA("Beam") or obj:IsA("Light") or obj:IsA("PostEffect")) and obj:GetAttribute("OrigEnabled") ~= nil then
        obj.Enabled = obj:GetAttribute("OrigEnabled")
    end
    if obj:IsA("Sky") then obj.Parent = lighting end -- Trả lại bầu trời
end

-- =======================================================
-- LÕI XỬ LÝ FIX LAG
-- =======================================================

local function ProcessObject(obj, level)
    SaveOriginalState(obj) -- LƯU LẠI TRƯỚC KHI XÓA/SỬA

    -- LEVEL 1: Xóa bóng và ánh sáng phụ
    if level >= 1 then
        if obj:IsA("BasePart") then obj.CastShadow = false
        elseif obj:IsA("Light") then obj.Enabled = false end
    end

    -- LEVEL 2: Tắt hiệu ứng chiêu thức (Particles)
    if level >= 2 then
        if obj:IsA("ParticleEmitter") or obj:IsA("Trail") or obj:IsA("Beam") or obj:IsA("Fire") or obj:IsA("Smoke") or obj:IsA("Sparkles") then
            obj.Enabled = false
        end
    end

    -- LEVEL 3: TÀN PHÁ ĐỒ HỌA (XÓA THÊM)
    if level >= 3 then
        if obj:IsA("BasePart") then
            obj.Material = Enum.Material.SmoothPlastic
            obj.Color = Color3.fromRGB(120, 120, 120) -- Màu xám xi măng
            obj.Transparency = 0.5 -- Làm mờ
        end
        if obj:IsA("MeshPart") then obj.TextureID = "" end
        if obj:IsA("SpecialMesh") then obj.TextureId = "" end
        if obj:IsA("Decal") or obj:IsA("Texture") then obj.Transparency = 1 end
        
        -- [MỚI] Xóa sạch phụ kiện (Accessories/Quần áo 3D) của NPC và Người chơi khác
        if obj:IsA("Accessory") or obj:IsA("ShirtGraphic") or obj:IsA("Clothing") then
            obj:Destroy() -- Cái này xóa hẳn cho nhẹ máy, ko cần khôi phục
        end
    end
end

-- Tối ưu Môi trường & Bầu trời
local function OptimizeEnvironment(level)
    if level >= 1 then
        lighting.GlobalShadows = false
    end
    if level >= 3 then
        -- [MỚI] Xóa bầu trời, xóa sương mù, xóa hiệu ứng
        lighting.FogEnd = 9e9
        for _, v in pairs(lighting:GetChildren()) do
            SaveOriginalState(v)
            if v:IsA("PostEffect") or v:IsA("Atmosphere") or v:IsA("ColorCorrectionEffect") or v:IsA("BloomEffect") or v:IsA("BlurEffect") or v:IsA("SunRaysEffect") then
                v.Enabled = false
            elseif v:IsA("Sky") then
                v.Parent = game.ReplicatedStorage -- Cất bầu trời đi thay vì xóa
            end
        end
        -- [MỚI] Xóa cỏ 3D
        if terrain then terrain.Decoration = false end
    end
end

-- =======================================================
-- HÀM QUÉT VÀ THỰC THI (CÓ CHỐNG ĐƠ MÁY)
-- =======================================================
local function RunOptimizer(level)
    CurrentLevel = level
    OptimizeEnvironment(level)
    
    local allObjects = workspace:GetDescendants()
    local total = #allObjects
    
    for i, obj in pairs(allObjects) do
        ProcessObject(obj, level)
        -- Cứ quét 1500 vật thể thì nghỉ 1 nhịp để máy không bị đơ/văng game
        if i % 1500 == 0 then task.wait() end 
    end
end

local function RunRestore()
    CurrentLevel = 0
    lighting.GlobalShadows = true
    if terrain then terrain.Decoration = true end -- Bật lại cỏ
    
    local allObjects = workspace:GetDescendants()
    for i, obj in pairs(allObjects) do
        RestoreObject(obj)
        if i % 1500 == 0 then task.wait() end 
    end
    
    -- Khôi phục đồ họa trong Lighting
    for _, v in pairs(lighting:GetChildren()) do RestoreObject(v) end
    for _, v in pairs(game.ReplicatedStorage:GetChildren()) do 
        if v:IsA("Sky") then v.Parent = lighting end 
    end
end

-- Tự động áp dụng cho các vật/chiêu thức mới sinh ra
workspace.DescendantAdded:Connect(function(obj)
    if CurrentLevel > 0 then
        -- Chờ 1 chút để vật thể load xong rồi mới fix
        task.wait(0.1) 
        ProcessObject(obj, CurrentLevel)
    end
end)
