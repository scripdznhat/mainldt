local dragging, dragInput, dragStart, startPos
TitleBar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = MainFrame.Position
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then dragging = false end
        end)
    end
end)
TitleBar.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then dragInput = input end
end)
UserInputService.InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        local delta = input.Position - dragStart
        MainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)

-- =======================================================
-- LÕI XỬ LÝ FIX LAG (CHUYÊN TRỊ BLOX FRUITS - SIÊU NHANH)
-- =======================================================
local CurrentLevel = 0
local isProcessing = false -- Khóa chống spam nút

local function OptimizeLightingAndTerrain()
    lighting.GlobalShadows = false
    lighting.FogEnd = 9e9
    lighting.Brightness = 1
    
    for _, v in pairs(lighting:GetChildren()) do
        if v:IsA("PostEffect") or v:IsA("Atmosphere") or v:IsA("Sky") or v:IsA("ColorCorrectionEffect") or v:IsA("BloomEffect") or v:IsA("BlurEffect") or v:IsA("SunRaysEffect") then
            v:Destroy()
        end
    end

    if terrain then
        terrain.WaterWaveSize = 0; terrain.WaterWaveSpeed = 0; terrain.WaterReflectance = 0; terrain.WaterTransparency = 1
        if CurrentLevel >= 2 then terrain.Decoration = false end
    end
end

local function ProcessObject(obj, level)
    if level >= 1 then
        if obj:IsA("BasePart") then obj.CastShadow = false
        elseif obj:IsA("PointLight") or obj:IsA("SurfaceLight") or obj:IsA("SpotLight") then obj.Enabled = false end
    end

    if level >= 2 then
        if obj:IsA("ParticleEmitter") or obj:IsA("Trail") or obj:IsA("Beam") or obj:IsA("Fire") or obj:IsA("Smoke") or obj:IsA("Sparkles") then
            obj.Enabled = false
            obj.Transparency = NumberSequence.new(1)
        end
    end

    if level >= 3 then
        if obj:IsA("BasePart") then
            obj.Material = Enum.Material.SmoothPlastic
            obj.Color = Color3.fromRGB(150, 150, 150)
            obj.Transparency = 0.5
            obj.Reflectance = 0
        end
        if obj:IsA("MeshPart") then obj.TextureID = ""
        elseif obj:IsA("SpecialMesh") then obj.TextureId = ""
        elseif obj:IsA("Decal") or obj:IsA("Texture") then obj.Transparency = 1
        elseif obj:IsA("Accessory") or obj:IsA("ShirtGraphic") or obj:IsA("Shirt") or obj:IsA("Pants") then
            obj:Destroy() 
        end
    end
end

-- TỐI ƯU HÓA HÀM CHẠY: Dùng task.spawn để không làm đơ máy
local function RunOptimizer(level, btn, doneText)
    if isProcessing then return end -- Đang chạy thì không cho bấm chồng lên
    isProcessing = true
    CurrentLevel = level
    OptimizeLightingAndTerrain()
    
    btn.Text = "Đang xử lý (1-2s)..."
    
    -- Mở một luồng chạy ngầm (Không làm đơ màn hình người chơi)
    task.spawn(function()
        local count = 0
        local function ScanFolder(folder)
            local objects = folder:GetDescendants()
            for i = 1, #objects do
                ProcessObject(objects[i], level)
                count = count + 1
                -- BÍ QUYẾT TĂNG TỐC: Cứ 5000 vật thể mới nghỉ một khung hình cực ngắn thay vì 1000
                if count >= 5000 then
                    count = 0
                    RunService.Heartbeat:Wait()
                end
            end
        end

        -- Quét Workspace
        ScanFolder(workspace)
        
        -- Quét ReplicatedStorage
        if level >= 2 then
            ScanFolder(replicatedStorage)
        end
        
        -- Khôi phục trạng thái nút
        btn.Text = doneText
        isProcessing = false
    end)
end

workspace.DescendantAdded:Connect(function(obj)
    if CurrentLevel > 0 then ProcessObject(obj, CurrentLevel) end
end)
